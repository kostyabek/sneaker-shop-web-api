CREATE TABLE `users_policies` (
  `id` int NOT NULL AUTO_INCREMENT,
  `userId` int NOT NULL,
  `policyId` int NOT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `id_UNIQUE` (`id`),
  KEY `userIdForpolicy_idx` (`userId`),
  KEY `policyIdForUser_idx` (`policyId`),
  CONSTRAINT `policyIdForUser` FOREIGN KEY (`policyId`) REFERENCES `policies` (`id`),
  CONSTRAINT `userIdForpolicy` FOREIGN KEY (`userId`) REFERENCES `users` (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=3 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci
